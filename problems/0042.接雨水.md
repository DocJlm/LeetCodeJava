# 42.接雨水
```
class Solution {
    public int trap(int[] height) {
        int ans = 0;
        int n = height.length;
        int l = 0, r = n - 1;

        int lMax = 0, rMax = 0;

        while (l < r) {
            lMax = Math.max(lMax, height[l]);
            rMax = Math.max(rMax, height[r]);

            if (height[l] < height[r]) {
                ans += lMax - height[l];
                l++;
            } else {
                ans += rMax - height[r];
                r--;
            }
        }
        return ans;
    }
}
```

# 239.滑动窗口最大值
```
class Solution {
    public int[] maxSlidingWindow(int[] nums, int k) {
        if (nums == null || nums.length == 0 || k <= 0) {
            return new int[0];
        }
        int n = nums.length;
        int[] result = new int[n - k + 1];
        int index = 0;
        // 双端队列，存储数组元素的索引，保证队列中元素对应的值单调递减
        Deque<Integer> deque = new LinkedList<>();
        
        for (int i = 0; i < n; i++) {
            // 移除队列中所有小于当前元素的元素索引，因为它们不可能成为后续窗口的最大值
            while (!deque.isEmpty() && nums[i] > nums[deque.peekLast()]) {
                deque.pollLast();
            }
            // 将当前元素索引加入队列
            deque.offerLast(i);
            
            // 移除队列头部超出窗口范围的元素索引
            if (deque.peekFirst() == i - k) {
                deque.pollFirst();
            }
            
            // 当窗口形成（i >= k - 1）时，记录当前窗口的最大值（队列头部元素）
            if (i >= k - 1) {
                result[index++] = nums[deque.peekFirst()];
            }
        }
        return result;
    }
}
```
